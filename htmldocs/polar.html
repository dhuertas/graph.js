Polar Plot

Similar to plot line, but now the grid shows angular increments. See example below:

Example code

var PI = Math.PI,
    cos = Math.cos,
    sin = Math.sin,
    pow = Math.pow;
		
var graph = new Graph({
    appendTo : "container",
    canvasWidth : 800,
    canvasHeight : 800/Math.sqrt(2),
    barPosition : "center",
    title : "Half-wave Dipole Antenna Radiation",
    yAxisTitle : "y",
    xAxisTitle : "x",
    polarAxisNumFormat : "deg",
    polarGridNumLines : 24
});

var px = [],
    py = [];

for (var theta = 0; theta <= 2*PI; theta += 0.01) {
    r = 1-pow(sin(theta), 2);
    r = r/2;

    px.push(r*cos(theta));
    py.push(r*sin(theta));
}

graph.polar(py, px);
