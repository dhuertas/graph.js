Plot Bars

Plot bar charts. See example below:

				var graph = new Graph({
					appendTo : "container",
					canvasWidth : 800,
					canvasHeight : 800/Math.sqrt(2),
					title : "Number of skipped lectures by months",
					xAxisTitle : "Months",
					yAxisTitle : "Skipped Lectures",
					colorList : ["#00F"],
					barWidth: 0.8,
					barPosition: "center",
					xAxisNumSteps:12,
					yAxisNumSteps: 10,
					yGridNumLines: 10,
					drawYAxis : 1,
					drawXAxis : 1
				});
				
			
				var px, py;
			
				py = [3,5,4,1,7,0,0,0,6,8,3,10];
				px = ["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"];
			
				graph.bars(py, px, false, [0,12]);
