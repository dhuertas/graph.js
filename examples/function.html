Plot Function

Plot a graph from a function. The library allows you to plot multiple lines in a same graph. See example below:

				var mean = 0, variance = 1;
		
				function Gaussian(x) {
					return Math.exp(-Math.pow(x-mean, 2)/(2*variance))/Math.sqrt(Math.PI*2*variance);
				}
		
				var graph = new Graph({
					appendTo : "container",
					canvasWidth : 800,
					canvasHeight : 800/Math.sqrt(2),
					title : "Gaussian Function",
					xAxisTitle : "x",
					yAxisTitle : "Gaussian(x)"
				});
			
				for (var i = 1; i < 6; i++) {
					variance = i;
					graph.plotFunction(Gaussian, [-7.5, 0.01, 7.5]);
				}
